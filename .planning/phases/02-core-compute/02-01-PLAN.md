    ```go
    func (c *VastAIClient) DeleteWithBody(ctx context.Context, path string, body, result interface{}) error {
        req, err := c.newRequest(ctx, http.MethodDelete, path, body)
        if err != nil {
            return err
        }
        return c.do(ctx, req, result)
    }
    ```
