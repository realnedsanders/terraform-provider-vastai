4. Create `internal/client/client.go`:
   - package client
   - Import: context, encoding/json, fmt, io, math, net/http, strconv, strings, time, github.com/hashicorp/go-retryablehttp, github.com/hashicorp/terraform-plugin-log/tflog
   - VastAIClient struct: httpClient *retryablehttp.Client, baseURL string, apiKey string, userAgent string
   - func NewVastAIClient(apiKey, baseURL, version string) *VastAIClient:
     - Create retryablehttp.NewClient()
     - Set RetryMax = 5 (per D-07)
     - Set RetryWaitMin = 150 * time.Millisecond (per D-07: 150ms base)
     - Set RetryWaitMax = 30 * time.Second
     - Set CheckRetry = vastaiRetryPolicy
     - Set Backoff = vastaiBackoff
     - Set Logger = nil (silence default; use tflog per Pitfall 6)
     - Strip trailing slash from baseURL with strings.TrimRight(baseURL, "/")
     - Set userAgent = fmt.Sprintf("terraform-provider-vastai/%s", version) (per FOUND-04)
   - func vastaiRetryPolicy(ctx context.Context, resp *http.Response, err error) (bool, error):
     - If ctx.Err() != nil: return false, ctx.Err()
     - If err != nil (connection error): return true, nil
     - If resp.StatusCode == 429: return true, nil (rate limited)
     - If resp.StatusCode >= 500 && resp.StatusCode != 501: return true, nil (server errors, not 501 Not Implemented)
     - Otherwise: return false, nil
   - func vastaiBackoff(min, max time.Duration, attemptNum int, resp *http.Response) time.Duration:
     - If resp != nil && resp.StatusCode == 429: check Retry-After header, parse as float seconds, return if valid
     - Calculate: wait = float64(min) * math.Pow(1.5, float64(attemptNum)) (per D-07: 1.5x multiplier)
     - If time.Duration(wait) > max: return max
     - Return time.Duration(wait)
   - func (c *VastAIClient) do(ctx context.Context, req *retryablehttp.Request, result interface{}) error:
     - Log request at DEBUG: tflog.Debug(ctx, "Vast.ai API request", map[string]interface{}{"method": req.Method, "url": req.URL.String()}) (per D-08)
     - Execute: resp, err := c.httpClient.Do(req)
     - If err != nil: return fmt.Errorf("executing request: %w", err)
     - defer resp.Body.Close()
     - Log response at DEBUG: tflog.Debug(ctx, "Vast.ai API response", map[string]interface{}{"status": resp.StatusCode}) (per D-08)
     - Read body: body, err := io.ReadAll(resp.Body)
     - Log body at TRACE: tflog.Trace(ctx, "Vast.ai API response body", map[string]interface{}{"body": string(body)}) (per D-08)
     - If resp.StatusCode >= 400: extract message from JSON body (try {"error": "..."} and {"msg": "..."} patterns), return &APIError{StatusCode: resp.StatusCode, Message: message, Method: req.Method, Path: req.URL.Path}
     - If result != nil && len(body) > 0: json.Unmarshal(body, result), return error if unmarshal fails
     - Return nil
   - func (c *VastAIClient) Get(ctx context.Context, path string, result interface{}) error:
     - req, err := c.newRequest(ctx, http.MethodGet, path, nil); if err != nil return
     - return c.do(ctx, req, result)
   - func (c *VastAIClient) Post(ctx context.Context, path string, body, result interface{}) error:
     - req, err := c.newRequest(ctx, http.MethodPost, path, body); if err != nil return
     - return c.do(ctx, req, result)
   - func (c *VastAIClient) Put(ctx context.Context, path string, body, result interface{}) error:
     - req, err := c.newRequest(ctx, http.MethodPut, path, body); if err != nil return
     - return c.do(ctx, req, result)
   - func (c *VastAIClient) Delete(ctx context.Context, path string, result interface{}) error:
     - req, err := c.newRequest(ctx, http.MethodDelete, path, nil); if err != nil return
     - return c.do(ctx, req, result)
